#include <stdio.h>

int main() {
    float pago, suma = 0.0;

    printf("Ingrese el primer pago:\t");
    scanf("%f", &pago);

    do {
        suma += pago;
        printf("Ingrese el siguiente pago (0 para terminar):\t");
        scanf("%f", &pago);
    } while (pago != 0);

    printf("El total de pagos del mes es: %.2f\n", suma);

    return 0;
}
