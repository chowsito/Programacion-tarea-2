#include <stdio.h>

int main() {
    int N, NUM, SUM = 0;
    printf("Ingrese el número de datos: ");
    scanf("%d", &N);

    for (int i = 1; i <= N; i++) {
        printf("Ingrese el dato número %d: ", i);
        scanf("%d", &NUM);

        if (NUM > 0) {
            SUM += NUM;
        }
    }

    printf("La suma de los números positivos es: %d\n", SUM);

    return 0;
}
