#include <stdio.h>

int main() {
    float PAG, SPA = 0;

    printf("Ingrese el primer pago: ");
    scanf("%f", &PAG);

    while (PAG != 0) {
        SPA += PAG;
        printf("Ingrese el siguiente pago (o 0 para terminar): ");
        scanf("%f", &PAG);
    }

    printf("El total de pagos del mes es: %.2f\n", SPA);

    return 0;
}
