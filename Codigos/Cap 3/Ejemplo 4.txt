#include <stdio.h>
#include <math.h>

int main() {
    int NUM;
    long CUA, SUC = 0;

    printf("Ingrese un número entero (0 para terminar): ");
    scanf("%d", &NUM);

    while (NUM != 0) {
        CUA = pow(NUM, 2);
        printf("%d al cuadrado es %ld\n", NUM, CUA);
        SUC += CUA;

        printf("Ingrese un número entero (0 para terminar): ");
        scanf("%d", &NUM);
    }

    printf("La suma de los cuadrados es %ld\n", SUC);

    return 0;
}
