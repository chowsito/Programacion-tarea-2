#include <stdio.h>

int main() {
    int numEmpleados;
    float salario, salarioTotal = 0, salarioPromedio;
    int contador = 0;

    printf("Ingrese el número de empleados de Amazon: ");
    scanf("%d", &numEmpleados);

    if (numEmpleados <= 0) {
        printf("El número de empleados debe ser mayor que cero.\n");
        return 1; // Salida con error
    }

    do {
        contador++;

        printf("Ingrese el salario del empleado %d: $", contador);
        scanf("%f", &salario);

        if (salario >= 0) {
            salarioTotal += salario;
        } else {
            printf("El salario no puede ser negativo. Ingrese un valor válido.\n");
            contador--; // Restar 1 al contador para repetir la entrada del salario
        }

    } while (contador < numEmpleados);

    salarioPromedio = salarioTotal / numEmpleados;

    printf("El salario total de los empleados de Amazon es: $%.2f\n", salarioTotal);
    printf("El salario promedio es: $%.2f\n", salarioPromedio);

    return 0;
}
