#include <stdio.h>

int main(void)
{
    float distancia, costo;
    int paquete;

    printf("Ingrese la distancia en kilómetros: ");
    scanf("%f", &distancia);

    printf("Ingrese el peso del paquete en kilogramos: ");
    scanf("%d", &paquete);

    if (distancia <= 50)
    {
        costo = 0.5 * distancia;
    }
    else if (distancia <= 150)
    {
        costo = 25 + 0.4 * (distancia - 50);
    }
    else
    {
        costo = 85 + 0.3 * (distancia - 150);
    }

    if (paquete > 10)
    {
        costo *= 1.2; // Aplicar un 20% de recargo por peso
    }

    printf("\nCosto de envío: $%.2f\n", costo);

    return 0;
}
